Replace loadProfiles function with this enhanced version
that handles photo_user_mirror linking:

async function loadProfiles(authUser) {
  setLoading(true)
  try {
    // 1. Try to find existing link in photo_user_mirror
    // Match by email (primary) or google_id (secondary)
    const googleId = authUser.user_metadata?.sub ||
                     authUser.user_metadata?.provider_id ||
                     null

    // Check by email first
    let { data: mirrorData } = await supabase
      .from('photo_user_mirror')
      .select('*')
      .eq('email', authUser.email)
      .maybeSingle()

    // If not found by email, try google_id
    if (!mirrorData && googleId) {
      const { data } = await supabase
        .from('photo_user_mirror')
        .select('*')
        .eq('google_id', googleId)
        .maybeSingle()
      mirrorData = data
    }

    if (mirrorData) {
      // Link auth_user_id if not yet linked
      if (!mirrorData.auth_user_id) {
        await supabase
          .from('photo_user_mirror')
          .update({
            auth_user_id: authUser.id,
            last_login: new Date().toISOString(),
            email_verified: true,
          })
          .eq('id', mirrorData.id)
      } else {
        // Just update last_login
        await supabase
          .from('photo_user_mirror')
          .update({ last_login: new Date().toISOString() })
          .eq('id', mirrorData.id)
      }
      setPhotoProfile({ ...mirrorData, auth_user_id: authUser.id })
    } else {
      // New user — insert into photo_user_mirror
      const newMirrorRow = {
        id: crypto.randomUUID(),
        email: authUser.email,
        name: authUser.user_metadata?.full_name ||
              authUser.user_metadata?.name ||
              authUser.email.split('@')[0],
        google_id: googleId,
        auth_user_id: authUser.id,
        email_verified: true,
        auth_type: 'google',
        last_login: new Date().toISOString(),
      }
      const { data: inserted } = await supabase
        .from('photo_user_mirror')
        .insert(newMirrorRow)
        .select()
        .single()
      setPhotoProfile(inserted)
    }

    // 2. Load or create my20fit_profile
    let { data: myProfile } = await supabase
      .from('my20fit_profile')
      .select('*')
      .eq('auth_user_id', authUser.id)
      .maybeSingle()

    if (!myProfile) {
      // Create profile using data from photo_user_mirror
      const displayName = mirrorData?.name ||
                          authUser.user_metadata?.full_name ||
                          authUser.user_metadata?.name ||
                          authUser.email.split('@')[0]

      const { data: created } = await supabase
        .from('my20fit_profile')
        .insert({
          auth_user_id: authUser.id,
          email: authUser.email,
          full_name: displayName,
          phone: mirrorData?.phone || null,
        })
        .select()
        .single()
      myProfile = created
    }

    setProfile(myProfile)

  } catch (err) {
    console.error('Error loading profiles:', err)
  } finally {
    setLoading(false)
  }
}

## STEP 4: UPDATE Login.jsx

GOOGLE OAUTH BUTTON (primary, most prominent):

async function handleGoogleLogin() {
  setGoogleLoading(true)
  setError(null)
  try {
    const { error } = await supabase.auth.signInWithOAuth({
      provider: 'google',
      options: {
        redirectTo: window.location.origin + '/',
        queryParams: {
          access_type: 'offline',
          prompt: 'consent',
        }
      }
    })
    if (error) setError(error.message)
  } catch (err) {
    setError('Gagal login dengan Google. Coba lagi.')
    setGoogleLoading(false)
  }
}

EMAIL LOGIN (secondary):

async function handleEmailLogin(email, password) {
  setLoading(true)
  setError(null)
  try {
    const { error } = await supabase.auth.signInWithPassword({
      email, password
    })
    if (error) {
      // Check if email exists in photo_user_mirror
      const { data: mirror } = await supabase
        .from('photo_user_mirror')
        .select('email, name, google_id')
        .eq('email', email)
        .maybeSingle()

      if (mirror?.google_id) {
        setError(
          'Akun ini terdaftar via Google. ' +
          'Gunakan tombol "Masuk dengan Google" di bawah.'
        )
      } else if (mirror) {
        setError('Password salah. Lupa password?')
      } else {
        setError('Email belum terdaftar. Silakan daftar terlebih dahulu.')
      }
    } else {
      navigate('/')
    }
  } finally {
    setLoading(false)
  }
}

EMAIL REGISTER:

async function handleEmailRegister(name, email, phone, password) {
  setLoading(true)
  setError(null)
  try {
    // Check if email already in photo_user_mirror
    const { data: existing } = await supabase
      .from('photo_user_mirror')
      .select('email, google_id, auth_user_id')
      .eq('email', email)
      .maybeSingle()

    // If existing user has google_id, suggest Google login
    if (existing?.google_id) {
      setError(
        'Email ini terdaftar di 20FIT via Google. ' +
        'Gunakan "Masuk dengan Google" untuk masuk.'
      )
      setLoading(false)
      return
    }

    // If already linked to Supabase Auth
    if (existing?.auth_user_id) {
      setError('Email ini sudah terdaftar. Silakan masuk.')
      setActiveTab('login')
      setLoading(false)
      return
    }

    // Register new user
    const { data, error } = await supabase.auth.signUp({
      email,
      password,
      options: {
        data: { full_name: name, phone }
      }
    })

    if (error) { setError(error.message); return }

    // Update phone in profile
    if (data.user) {
      await supabase
        .from('my20fit_profile')
        .update({ phone, full_name: name })
        .eq('auth_user_id', data.user.id)
    }

    navigate('/')

  } finally {
    setLoading(false)
  }
}

## STEP 8: RLS POLICY FOR photo_user_mirror

Run this SQL in Supabase SQL Editor:

-- Allow authenticated users to read their own mirror row
CREATE POLICY "Users can read own mirror"
ON photo_user_mirror FOR SELECT
USING (auth_user_id = auth.uid() OR email = auth.jwt()->>'email');

-- Allow authenticated users to update their own mirror row
CREATE POLICY "Users can update own mirror"
ON photo_user_mirror FOR UPDATE
USING (auth_user_id = auth.uid() OR email = auth.jwt()->>'email');

-- Allow insert for new users
CREATE POLICY "Users can insert own mirror"
ON photo_user_mirror FOR INSERT
WITH CHECK (auth_user_id = auth.uid());

-- Enable RLS if not already enabled
ALTER TABLE photo_user_mirror ENABLE ROW LEVEL SECURITY;
