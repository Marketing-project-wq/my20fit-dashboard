## STEP 3: AUTH CONTEXT

Create src/contexts/AuthContext.jsx:

import { createContext, useContext, useEffect, useState } from 'react'
import { supabase } from '../lib/supabase'

const AuthContext = createContext({})

export function AuthProvider({ children }) {
  const [user, setUser] = useState(null)
  const [profile, setProfile] = useState(null)
  const [photoProfile, setPhotoProfile] = useState(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    // Get initial session
    supabase.auth.getSession().then(({ data: { session } }) => {
      setUser(session?.user ?? null)
      if (session?.user) loadProfiles(session.user)
      else setLoading(false)
    })

    // Listen for auth changes
    const { data: { subscription } } = supabase.auth.onAuthStateChange(
      async (event, session) => {
        setUser(session?.user ?? null)
        if (session?.user) await loadProfiles(session.user)
        else {
          setProfile(null)
          setPhotoProfile(null)
          setLoading(false)
        }
      }
    )
    return () => subscription.unsubscribe()
  }, [])

  async function loadProfiles(authUser) {
    try {
      // Load my20fit profile
      const { data: myProfile } = await supabase
        .from('my20fit_profile')
        .select('*')
        .eq('auth_user_id', authUser.id)
        .single()
      setProfile(myProfile)

      // Load photo_user_mirror profile
      const { data: photoData } = await supabase
        .from('photo_user_mirror')
        .select('*')
        .eq('auth_user_id', authUser.id)
        .single()
      setPhotoProfile(photoData)
    } finally {
      setLoading(false)
    }
  }

  async function signOut() {
    await supabase.auth.signOut()
    setUser(null)
    setProfile(null)
    setPhotoProfile(null)
  }

  const value = { user, profile, photoProfile, loading, signOut, loadProfiles }
  return <AuthContext.Provider value={value}>{children}</AuthContext.Provider>
}

export const useAuth = () => useContext(AuthContext)

## STEP 5: AUTH LOGIC IN Login.jsx

### EMAIL LOGIN:

async function handleLogin(email, password) {
  setLoading(true)
  setError(null)
  
  try {
    const { data, error } = await supabase.auth.signInWithPassword({
      email,
      password
    })
    
    if (error) {
      // Check if user exists in photo_user_mirror but not in auth
      if (error.message.includes('Invalid login credentials')) {
        const { data: mirror } = await supabase
          .from('photo_user_mirror')
          .select('email, name, auth_type')
          .eq('email', email)
          .single()
        
        if (mirror) {
          setError('Email ini terdaftar di 20FIT. Gunakan password yang sama dengan akun 20FIT kamu, atau reset password.')
        } else {
          setError('Email atau password salah.')
        }
      } else {
        setError(error.message)
      }
      return
    }
    
    // Success → navigate to dashboard
    navigate('/')
    
  } finally {
    setLoading(false)
  }
}

### EMAIL REGISTER:

async function handleRegister(name, email, phone, password) {
  setLoading(true)
  setError(null)
  
  try {
    // Check if email exists in photo_user_mirror
    const { data: existing } = await supabase
      .from('photo_user_mirror')
      .select('email, name, auth_user_id')
      .eq('email', email)
      .single()
    
    if (existing?.auth_user_id) {
      setError('Email ini sudah terdaftar. Silakan masuk.')
      setActiveTab('login')
      return
    }
    
    // Register with Supabase Auth
    const { data, error } = await supabase.auth.signUp({
      email,
      password,
      options: {
        data: {
          full_name: name,
          phone: phone
        }
      }
    })
    
    if (error) { setError(error.message); return }
    
    // Trigger function handles linking to photo_user_mirror automatically
    // Update phone in my20fit_profile
    if (data.user) {
      await supabase
        .from('my20fit_profile')
        .update({ phone, full_name: name })
        .eq('auth_user_id', data.user.id)
    }
    
    // Navigate to dashboard
    navigate('/')
    
  } finally {
    setLoading(false)
  }
}

async function handleGoogleLogin() {
  const { error } = await supabase.auth.signInWithOAuth({
    provider: 'google',
    options: {
      redirectTo: window.location.origin + '/'
    }
  })
  if (error) setError(error.message)
}

async function handleForgotPassword(email) {
  const { error } = await supabase.auth.resetPasswordForEmail(email, {
    redirectTo: window.location.origin + '/reset-password'
  })
  if (error) setError(error.message)
  else setForgotSuccess(true)
}

## STEP 6: PROTECTED ROUTES

Update App.jsx routing:

import { Navigate } from 'react-router-dom'
import { useAuth } from './contexts/AuthContext'

function ProtectedRoute({ children }) {
  const { user, loading } = useAuth()
  
  if (loading) return (
    <div style={{
      display: 'flex', alignItems: 'center', justifyContent: 'center',
      height: '100vh', background: 'var(--bg)'
    }}>
      <div style={{
        width: 40, height: 40, borderRadius: '50%',
        border: '3px solid #C41101',
        borderTopColor: 'transparent',
        animation: 'spin 0.7s linear infinite'
      }}/>
    </div>
  )
  
  if (!user) return <Navigate to="/login" replace />
  return children
}
