1. Initial state — all items unchecked:

   const loadChecklist = () => {
     const saved = localStorage.getItem('my20fit_mcu_result')
     if (!saved) return []
     const mcu = JSON.parse(saved)
     return (mcu.checklist || []).map(item => ({
       ...item,
       done: false  // always start unchecked
     }))
   }

2. Daily reset logic — checklist resets every new day:
   On component mount, check if today's date matches 
   localStorage 'my20fit_checklist_date':
   
   useEffect(() => {
     const today = new Date().toISOString().split('T')[0] // YYYY-MM-DD
     const savedDate = localStorage.getItem('my20fit_checklist_date')
     
     if (savedDate !== today) {
       // New day — reset all to unchecked
       localStorage.setItem('my20fit_checklist_date', today)
       setItems(prev => prev.map(item => ({ ...item, done: false })))
       // Reset all-done header state too
       setAllDone(false)
     } else {
       // Same day — load saved check states
       const savedStates = localStorage.getItem('my20fit_checklist_states')
       if (savedStates) {
         const states = JSON.parse(savedStates)
         setItems(prev => prev.map((item, i) => ({
           ...item,
           done: states[i] ?? false
         })))
       }
     }
   }, [])

   const toggleItem = (index) => {
