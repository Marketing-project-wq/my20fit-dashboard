// Read all sleep data from localStorage (last N days)
export function getSleepHistory(days = 30) {
  const result = []
  for (let i = 0; i < days; i++) {
    const date = new Date()
    date.setDate(date.getDate() - i)
    const key = `my20fit_sleep_${date.toISOString().split('T')[0]}`
    const saved = localStorage.getItem(key)
    if (saved) {
      try {
        const data = JSON.parse(saved)
        // data has bedTime and wakeTime
        const [bH, bM] = data.bedTime.split(':').map(Number)
        const [wH, wM] = data.wakeTime.split(':').map(Number)
        let mins = (wH*60+wM) - (bH*60+bM)
        if (mins < 0) mins += 24*60
        const hours = parseFloat((mins/60).toFixed(1))
        result.push({
          date: date.toISOString().split('T')[0],
          hours,
          bedTime: data.bedTime,
          wakeTime: data.wakeTime,
          quality: data.quality || null
        })
      } catch {}
    }
  }
  return result.reverse() // oldest first
}

// Read all water data from localStorage (last N days)
export function getWaterHistory(days = 30) {
  const result = []
  for (let i = 0; i < days; i++) {
    const date = new Date()
    date.setDate(date.getDate() - i)
    const dateStr = date.toISOString().split('T')[0]
    const key = `my20fit_water_${dateStr}`
    const saved = localStorage.getItem(key)
    if (saved) {
      try {
        const data = JSON.parse(saved)
        const totalMl = (data.logs || []).reduce((sum, l) => sum + l.ml, 0)
        const totalL = parseFloat((totalMl / 1000).toFixed(2))
        result.push({
          date: dateStr,
          liters: totalL,
          cups: (data.logs || []).reduce((sum, l) => sum + l.cups, 0),
          target: data.weight ? parseFloat((data.weight * 0.033).toFixed(1)) : 2.5,
          pct: data.weight
            ? Math.round((totalL / (data.weight * 0.033)) * 100)
            : Math.round((totalL / 2.5) * 100)
        })
      } catch {}
    }
  }
  return result.reverse()
}

// Read all wellness data from localStorage (last N days)
export function getWellnessHistory(days = 30) {
  const result = []
  for (let i = 0; i < days; i++) {
    const date = new Date()
    date.setDate(date.getDate() - i)
    const dateStr = date.toISOString().split('T')[0]
    const key = `my20fit_wellness_${dateStr}`
    const saved = localStorage.getItem(key)
    if (saved) {
      try {
        const data = JSON.parse(saved)
        result.push({
          date: dateStr,
          mood: data.mood || null,
          energy: data.energy || null,
          stress: data.stress || null,
          soreness: data.soreness || null
        })
      } catch {}
    }
  }
  return result.reverse()
}

// Read checklist completion history
export function getChecklistHistory(days = 30) {
  const result = []
  for (let i = 0; i < days; i++) {
    const date = new Date()
    date.setDate(date.getDate() - i)
    const dateStr = date.toISOString().split('T')[0]
    // Check if checklist_date matches and get states
    const savedDate = localStorage.getItem('my20fit_checklist_date')
    const savedStates = localStorage.getItem('my20fit_checklist_states')
    if (savedDate === dateStr && savedStates) {
      const states = JSON.parse(savedStates)
      const completed = states.filter(Boolean).length
      const total = states.length
      result.push({ date: dateStr, completed, total, pct: Math.round((completed/total)*100) })
    }
  }
  return result.reverse()
}

## FEATURE 2: TREND INDICATORS

### Trend Calculation Utility

Add to src/utils/checkinData.js:

export function calculateTrend(dataArray, valueKey) {
  if (!dataArray || dataArray.length < 2) return null
  
  const sorted = [...dataArray].sort((a,b) => 
    new Date(a.date) - new Date(b.date)
  )
  
  const latest = sorted[sorted.length - 1][valueKey]
  const previous = sorted[sorted.length - 2][valueKey]
  
  if (!latest || !previous) return null
  
  const diff = latest - previous
  const pct = Math.abs(Math.round((diff / previous) * 100))
  
  return {
    direction: diff > 0 ? 'up' : diff < 0 ? 'down' : 'same',
    diff: Math.abs(diff).toFixed(1),
    pct,
    latest,
    previous
  }
}

export function getMultiPointTrend(dataArray, valueKey, points = 7) {
  if (!dataArray || dataArray.length < 2) return null
  
  const sorted = [...dataArray]
    .sort((a,b) => new Date(a.date) - new Date(b.date))
    .slice(-points)
  
  if (sorted.length < 2) return null
  
  const first = sorted[0][valueKey]
  const last = sorted[sorted.length-1][valueKey]
  const diff = last - first
  const pct = Math.abs(Math.round((diff/first)*100))
  
  // Linear regression for trend direction
  const n = sorted.length
  const xMean = (n-1)/2
  const yMean = sorted.reduce((s,d) => s + d[valueKey], 0) / n
  const slope = sorted.reduce((s,d,i) => 
    s + (i - xMean) * (d[valueKey] - yMean), 0
  ) / sorted.reduce((s,_,i) => s + Math.pow(i-xMean,2), 0)
  
  return {
    direction: slope > 0.05 ? 'up' : slope < -0.05 ? 'down' : 'stable',
    diff: Math.abs(diff).toFixed(1),
    pct,
    slope: slope.toFixed(3)
  }
}

### Trend Indicator Component

Create src/components/TrendBadge.jsx:

export default function TrendBadge({ trend, metric, positiveIsGood = true }) {
  if (!trend || trend.direction === 'same') return (
    <span style={{
      display:'inline-flex', alignItems:'center', gap:3,
      fontFamily:"'Barlow Condensed'",
      fontSize:12, color:'var(--muted)'
    }}>
      <Minus size={12}/> Stabil
    </span>
  )

  // Determine if this trend is good or bad
  // e.g. weight up = bad, sleep up = good, stress up = bad
  const isGood = positiveIsGood
    ? trend.direction === 'up'
    : trend.direction === 'down'

  const color = isGood ? '#22C55E' : '#EF4444'
  const Icon = trend.direction === 'up' ? TrendingUp : TrendingDown

  return (
    <span style={{
      display:'inline-flex', alignItems:'center', gap:4,
      background: color + '15',
      color: color,
      padding:'3px 8px', borderRadius:99,
      fontFamily:"'Barlow Condensed'",
      fontSize:12, fontWeight:600
    }}>
      <Icon size={12}/>
      {trend.direction === 'up' ? '+' : '-'}{trend.diff}
      <span style={{ opacity:.7 }}>({trend.pct}%)</span>
    </span>
  )
}
